1// Package httpapi zawiera lokalne API HTTP klienta.
2//
3// /Pakiet udostępnia endpointy używane przez interfejs użytkownika do uruchamiania
4// /uwierzytelniania oraz wysyłania wiadomości do serwera po zestawieniu sesji.
5package httpapi
6
7import "net/http"
8
9// ReadMessageUseCase definiuje operację wysyłania wiadomości do serwera.
10//
11// /Interfejs oddziela warstwę HTTP od logiki aplikacyjnej odpowiedzialnej
12// /za szyfrowanie wiadomości, komunikację z serwerem i odszyfrowanie odpowiedzi.
13type ReadMessageUseCase interface {
14 ReadMessage(msg string) (string, error)
15}
16
17// AuthenticateUseCase definiuje operację uwierzytelniania klienta.
18//
19// /Implementacja tego interfejsu wykonuje proces uwierzytelnienia z udziałem
20// /klienta, serwera oraz TTP, a po sukcesie zapisuje lokalny klucz sesyjny.
21type AuthenticateUseCase interface {
22 Authenticate() error
23}
24
25// Server reprezentuje lokalny serwer HTTP klienta.
26//
27// /Struktura przechowuje zależności do przypadków użycia oraz katalog bazowy,
28// /w którym zapisywany jest lokalny stan klienta, między innymi klucz sesyjny.
29type Server struct {
30 readMessage ReadMessageUseCase
31 authenticate AuthenticateUseCase
32 baseDir string
33}
34
35// New tworzy nową instancję lokalnego serwera HTTP klienta.
36//
37// /Funkcja przyjmuje implementacje przypadków użycia oraz katalog bazowy
38// /lokalnej tożsamości, a następnie zwraca gotową strukturę Server.
39//
40// /@param readMessage Przypadek użycia odpowiedzialny za wysyłanie wiadomości.
41// /@param authenticate Przypadek użycia odpowiedzialny za uwierzytelnianie klienta.
42// /@param baseDir Katalog bazowy lokalnej tożsamości klienta.
43// /@return Wskaźnik do nowej instancji Server.
44func New(
45 readMessage ReadMessageUseCase,
46 authenticate AuthenticateUseCase,
47 baseDir string,
48) *Server {
49 return &Server{
50 readMessage: readMessage,
51 authenticate: authenticate,
52 baseDir: baseDir,
53 }
54}
55
56// Handler buduje główny router HTTP lokalnego API klienta.
57//
58// /Funkcja rejestruje endpoint sprawdzający stan aplikacji, endpoint wysyłania
59// /wiadomości oraz endpoint uruchamiający proces uwierzytelniania.
60//
61// /@return Handler HTTP gotowy do przekazania do serwera net/http.
62func (s *Server) Handler() http.Handler {
63 mux := http.NewServeMux()
64
65 mux.HandleFunc("/health", s.handleHealth)
66 mux.HandleFunc("/api/message", s.handleMessage)
67 mux.HandleFunc("/api/authenticate", s.handleAuthenticate)
68
69 return mux
70}
