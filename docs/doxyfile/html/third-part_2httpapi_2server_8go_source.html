1// Package httpapi zawiera API HTTP aplikacji TTP.
2//
3// /Pakiet definiuje interfejs logiki TTP, strukturę serwera HTTP,
4// /konstruktor oraz routing endpointów wykorzystywanych podczas inicjalizacji,
5// /rejestracji i uwierzytelniania.
6package httpapi
7
8import (
9 "net/http"
10
11 "scs/internal/shared/protocol"
12)
13
14// TTPService definiuje operacje udostępniane przez usługę Trusted Third Party.
15//
16// /Interfejs oddziela warstwę HTTP od logiki domenowej TTP. Implementacja
17// /odpowiada za udostępnienie publicznego klucza TTP, rejestrację stron,
18// /wydawanie certyfikatów oraz uwierzytelnianie klienta i serwera.
19type TTPService interface {
20 Init() (protocol.InitResponse, error)
21 Register(req protocol.RegisterRequest) (protocol.RegisterResponse, error)
22 Authenticate(req protocol.AuthenticateRequest) (protocol.AuthenticateResponse, error)
23}
24
25// Server reprezentuje serwer HTTP aplikacji TTP.
26//
27// /Struktura przechowuje zależność do logiki TTP, która jest wywoływana
28// /przez handlery HTTP obsługujące poszczególne etapy protokołu.
29type Server struct {
30 ttp TTPService
31}
32
33// New tworzy nową instancję serwera HTTP TTP.
34//
35// /Funkcja przyjmuje implementację usługi TTP i opakowuje ją w warstwę HTTP,
36// /dzięki czemu logika zaufanej strony trzeciej może być dostępna przez REST API.
37//
38// /@param ttp Implementacja logiki Trusted Third Party.
39// /@return Wskaźnik do nowej instancji Server.
40func New(ttp TTPService) *Server {
41 return &Server{
42 ttp: ttp,
43 }
44}
45
46// Handler buduje główny router HTTP aplikacji TTP.
47//
48// /Funkcja rejestruje endpointy sprawdzania stanu, inicjalizacji,
49// /rejestracji klienta lub serwera oraz uwierzytelniania stron.
50//
51// /@return Handler HTTP gotowy do przekazania do serwera net/http.
52func (s *Server) Handler() http.Handler {
53 mux := http.NewServeMux()
54
55 mux.HandleFunc("/health", s.handleHealth)
56 mux.HandleFunc("/api/init", s.handleInit)
57 mux.HandleFunc("/api/register", s.handleRegister)
58 mux.HandleFunc("/api/authenticate", s.handleAuthentication)
59
60 return mux
61}
