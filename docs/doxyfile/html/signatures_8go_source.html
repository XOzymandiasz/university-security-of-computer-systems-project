12//
13// /Funkcja oblicza skrót SHA-256 z przekazanych danych, a następnie podpisuje

16//
17// /@param data Dane, które mają zostać podpisane.
18// /@param privateKey Klucz prywatny RSA używany do utworzenia podpisu.
19// /@return Podpis cyfrowy w formacie Base64 lub błąd podpisywania.
20func SignBase64(data []byte, privateKey *rsa.PrivateKey) (string, error) {
21 digest := sha256.Sum256(data)
22
23 signature, err := rsa.SignPSS(
24 rand.Reader,
25 privateKey,
26 crypto.SHA256,
27 digest[:],
28 nil,
29 )
30 if err != nil {
31 return "", err
32 }
33
34 return base64.StdEncoding.EncodeToString(signature), nil
35}
