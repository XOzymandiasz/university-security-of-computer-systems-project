12// SaveSessionKey zapisuje klucz sesyjny AES w lokalnym magazynie aplikacji.
13//
14// /Funkcja tworzy katalog bazowy, jeśli jeszcze nie istnieje, koduje klucz sesyjny
15// /do formatu Base64 i zapisuje go w pliku. Klucz sesyjny jest używany po poprawnym
16// /uwierzytelnieniu klienta i serwera przez TTP.
17//
18// /@param baseDir Katalog bazowy, w którym ma zostać zapisany klucz sesyjny.
19// /@param sessionKey Klucz sesyjny AES otrzymany od TTP.
20// /@return Błąd zapisu lub nil w przypadku powodzenia.
21func SaveSessionKey(baseDir string, sessionKey []byte) error {
22 if err := os.MkdirAll(baseDir, 0700); err != nil {
23 return err
24 }
25
26 encoded := base64.StdEncoding.EncodeToString(sessionKey)
27
28 return os.WriteFile(filepath.Join(baseDir, sessionFileName), []byte(encoded), 0600)
29}
