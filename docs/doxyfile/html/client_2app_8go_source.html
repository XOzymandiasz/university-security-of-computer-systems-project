58// Bootstrap przygotowuje klienta do pracy w protokole TTP.
59//
60// /Funkcja pobiera publiczny klucz szyfrujący TTP, tworzy lokalną tożsamość
61// /klienta, szyfruje identyfikator klienta kluczem publicznym TTP, rejestruje
62// /klienta w TTP oraz zapisuje otrzymany certyfikat X.509.
63//
64// /@return Błąd inicjalizacji lub nil w przypadku powodzenia.
65func (a *App) Bootstrap() error {
66 ttpPublicKey, err := a.ttpClient.Init()
67 if err != nil {
68 return fmt.Errorf("ttp init: %w", err)
69 }
70
71 err = identity.EnsureIdentity(a.config.BaseDir)
72 if err != nil {
73 return err
74 }
75
76 var data protocol.RegisterRequest
77 data, err = identity.LoadRegistrationData(a.config.BaseDir)
78
79 if err != nil {
80 return fmt.Errorf("load registration data: %w", err)
81 }
82
83 var encryptedID string
84 encryptedID, err = identity.EncryptWithPublicKeyBase64([]byte(data.EncryptedID), ttpPublicKey)
85 if err != nil {
86 return fmt.Errorf("encrypt id for ttp: %w", err)
87 }
88
89 var certificateBase64 string
90 certificateBase64, err = a.ttpClient.Register(
91 protocol.RegisterRequest{
92 EncryptedID: encryptedID,
93 EncPublicKey: data.EncPublicKey,
94 AuthPublicKey: data.AuthPublicKey,
95 Role: protocol.EntityRoleClient,
96 })
97
98 if err != nil {
99 return fmt.Errorf("ttp register: %w", err)
100 }
101
102 if err = identity.SaveCertificate(a.config.BaseDir, certificateBase64); err != nil {
103 return fmt.Errorf("save certificate: %w", err)
104 }
105
106 return nil
107}
